@Pipe({
  name: 'weatherNameToIcon'
})
export class WeatherNameToIconPipe implements PipeTransform {

  transform(id: any, args?: any): any {
    let code: string = 'wi-owm-' + id;
    let foundKey = findKey(OWMLIST, (o: any) => { return (o === id); });

    if (foundKey) {
      return foundKey;
    } else if (OWMLIST[code]) {
      return code;
    }
    return null;
  }

}
